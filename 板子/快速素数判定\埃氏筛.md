可快速判定N<=1e12的素数
```C++
const int sqrtN=1e6+5;
bool prime[sqrtN];
vector<long long> p;
void init_prime(){//埃氏筛
	memset(prime,true,sizeof(prime));
	for(int i=2;i*i<=sqrtN;i++){
		if(prime[i]){
			for(int j=2*i;j<sqrtN;j+=i){
				prime[j]=false;
			}
		}
	}
	for(int i=2;i<sqrtN;i++){
		if(prime[i]){
			p.push_back(i);
		}
	}
}
bool is_prime(long long num){
	for(int i=0;i<p.size()&&p[i]<num;i++){
		if(num%p[i]==0){
			return false;
		}
	}
	return true;
}
```
